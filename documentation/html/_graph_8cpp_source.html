104double Graph::calculateDistance(Airport *a1, Airport *a2) {
105 double lat1 = a1->getLatitude();
106 double lat2 = a2->getLatitude();
107 double lon1 = a1->getLongitude();
108 double lon2 = a2->getLongitude();
109 int earthRadius = 6371000; // metres
110 double latitudeSource = lat1 * M_PI / 180;
111 double latitudeDestination = lat2 * M_PI / 180;
112 double latitudeDiff = (lat2 - lat1) * M_PI / 180;
113 double longitudeDiff = (lon2 - lon1) * M_PI / 180;
114
115 double a = sin(latitudeDiff / 2) * sin(latitudeDiff / 2) +
116 cos(latitudeSource) * cos(latitudeDestination) *
117 sin(longitudeDiff / 2) * sin(longitudeDiff / 2);
118
119 double c = 2 * atan2(sqrt(a), sqrt(1 - a));
120
121 return earthRadius * c / 1000; // in km
122}

143 q.push(w);

149 curr.first = u;
150 curr.second = e.getAirline();
151 current.push_back(curr);
152 w->setScales(current);
153 }
154 if (w == destination)
